# =============================================================================
# Generic PII patterns — public, no client- or person-specific values.
#
# Format: one regex per non-blank, non-comment line. Pattern is compiled with
# re.MULTILINE; case sensitivity is the default (use (?i) inline if you need
# case-insensitive). Lines beginning with `#` are comments.
#
# DO NOT add specific names, client identifiers, or personal handles here.
# Personal patterns live in .pii-rules.local (gitignored).
# =============================================================================

# -----------------------------------------------------------------------------
# Email addresses (any non-whitespace local + @ + domain + TLD)
# -----------------------------------------------------------------------------
\S+@\S+\.\w+

# -----------------------------------------------------------------------------
# Credentials and access tokens
# -----------------------------------------------------------------------------

# AWS access keys
AKIA[0-9A-Z]{16}

# AWS secret access keys (40-char base64-ish; high false-positive risk if you
# embed it more loosely, so keep it tight to the AWS pattern)
aws_secret_access_key\s*=\s*['"][A-Za-z0-9/+=]{40}['"]

# GitHub PATs
ghp_[A-Za-z0-9]{36}
gho_[A-Za-z0-9]{36}
ghu_[A-Za-z0-9]{36}
ghs_[A-Za-z0-9]{36}
ghr_[A-Za-z0-9]{36}
github_pat_[A-Za-z0-9_]{82}

# OpenAI keys
sk-[A-Za-z0-9]{48}
sk-proj-[A-Za-z0-9_-]{43,}

# Anthropic keys
sk-ant-[A-Za-z0-9_-]{20,}

# Slack tokens
xox[baprs]-[A-Za-z0-9-]+

# Discord tokens (mixed case, three dot-separated base64 segments)
[MN][A-Za-z\d]{23}\.[A-Za-z\d_-]{6}\.[A-Za-z\d_-]{27}

# Google API keys
AIza[0-9A-Za-z_-]{35}

# Stripe live keys
sk_live_[A-Za-z0-9]{24,}
rk_live_[A-Za-z0-9]{24,}

# JWT (header.payload.signature)
eyJ[A-Za-z0-9_-]{10,}\.eyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}

# -----------------------------------------------------------------------------
# Cryptographic material
# -----------------------------------------------------------------------------

# PEM private key headers (RSA / EC / generic / OpenSSH)
-----BEGIN [A-Z ]+PRIVATE KEY-----
-----BEGIN OPENSSH PRIVATE KEY-----
-----BEGIN PGP PRIVATE KEY BLOCK-----

# -----------------------------------------------------------------------------
# Local filesystem leakage
# -----------------------------------------------------------------------------

# Generic Windows user paths — catches any \Users\<name>\ leak (case-insensitive
# drive letter + slash variants). Safer than naming any specific username.
[Cc]:[/\\]Users[/\\]\w+

# Generic macOS user paths
/Users/[a-zA-Z][a-zA-Z0-9._-]+/

# Generic Linux home paths (limit to common shells/dotfiles to avoid matching
# every `/home/...` reference in docs)
/home/[a-z][a-z0-9_-]*/\.

# -----------------------------------------------------------------------------
# Common environment-variable secret references when checked in by accident
# -----------------------------------------------------------------------------
ANTHROPIC_API_KEY\s*=\s*sk-ant-
OPENAI_API_KEY\s*=\s*sk-
GITHUB_TOKEN\s*=\s*gh[ousprs]_
